Update a field of a table with an estimate of “Moradores em domicílios particulares permanentes” per square kilometer using area weighted interpolation.

WITH data AS (
  SELECT cartodb_id AS id, the_geom FROM <table_to_update>
),
meta AS (
  SELECT OBS_GetMeta(ST_SetSRID(ST_Extent(the_geom), 4326),
  '[{"numer_id": "br.data.Entorno03_V422",
    "normalization": "areaNormalized"}]'
  , 1, 1) meta
FROM data)
UPDATE <table_to_update> SET <field_to_update> = (data->0->>'value')::Numeric
FROM OBS_GetData(
      (SELECT ARRAY_AGG((the_geom, id)::geomval) FROM data),
      (SELECT meta FROM meta)) obs_data
WHERE <table_to_update>.cartodb_id = obs_data.id

Obtain an estimate of “Moradores em domicílios particulares permanentes” per square kilometer using area weighted interpolation.

WITH data AS (
  SELECT 1 AS id,
  ST_Buffer(ST_SetSRID(ST_Point(-22.9, -43.19), 4326), 0.01) the_geom
),
meta AS (
  SELECT OBS_GetMeta(ST_SetSRID(ST_Extent(the_geom), 4326),
  '[{"numer_id": "br.data.Entorno03_V422",
     "normalization": "areaNormalized"}]'
  , 1, 1) meta
FROM data)
SELECT id, (data->0->>'value')::Numeric AS entorno03_v422_per_sq_km_2010
FROM OBS_GetData(
       (SELECT ARRAY_AGG((the_geom, id)::geomval) FROM data),
       (SELECT meta FROM meta))
