Employment level location quotient in Financial activities establishments¶

WITH data AS (
  SELECT cartodb_id AS id, the_geom FROM <table_to_update>
),
meta AS (
  SELECT OBS_GetMeta(ST_SetSRID(ST_Extent(the_geom), 4326),
  '[{"numer_id": "us.bls.lq_month3_emplvl_1023"}]'
  , 1, 1) meta
FROM data)
UPDATE <table_to_update> SET <field_to_update> = (data->0->>'value')::Numeric
FROM OBS_GetData(
      (SELECT ARRAY_AGG((the_geom, id)::geomval) FROM data),
      (SELECT meta FROM meta)) obs_data
WHERE <table_to_update>.cartodb_id = obs_data.id

Obtain the value of “Employment level location quotient in Financial activities establishments” for a point. The value will be from the densest boundary with this measure.

WITH data AS (
  SELECT 1 AS id,
  ST_SetSRID(ST_Point(-73.9, 40.7), 4326) the_geom
),
meta AS (
  SELECT OBS_GetMeta(ST_SetSRID(ST_Extent(the_geom), 4326),
  '[{"numer_id": "us.bls.lq_month3_emplvl_1023"}]'
  , 1, 1) meta
FROM data)
SELECT id, (data->0->>'value')::Numeric AS lq_month3_emplvl_1023_financial_activities
FROM OBS_GetData(
       (SELECT ARRAY_AGG((the_geom, id)::geomval) FROM data),
       (SELECT meta FROM meta))
