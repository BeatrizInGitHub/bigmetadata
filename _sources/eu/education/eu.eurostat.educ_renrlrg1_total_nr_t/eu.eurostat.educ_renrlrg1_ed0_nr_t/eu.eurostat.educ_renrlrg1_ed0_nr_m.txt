      Update a field of a table with an estimate of "Students - Pre-primary education (level 0), Males" per Students - Pre-primary education (level 0), Total for a polygon
      using a combination of weighting by area and Students - Pre-primary education (level 0), Total.

      .. code-block:: postgresql

          WITH data AS (
            SELECT cartodb_id AS id, the_geom FROM <table_to_update>
          ),
          meta AS (
            SELECT OBS_GetMeta(ST_SetSRID(ST_Extent(the_geom), 4326),
            '[{"numer_id": "eu.eurostat.educ_renrlrg1_ed0_nr_m",
              "denom_id": "eu.eurostat.educ_renrlrg1_ed0_nr_t",
              "normalization": "denominated"}]'
            , 1, 1) meta
          FROM data)
          UPDATE <table_to_update> SET <field_to_update> = (data->0->>'value')::Numeric
          FROM OBS_GetData(
                (SELECT ARRAY_AGG((the_geom, id)::geomval) FROM data),
                (SELECT meta FROM meta)) obs_data
          WHERE <table_to_update>.cartodb_id = obs_data.id
