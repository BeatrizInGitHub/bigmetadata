------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------

3e décile du revenu déclaré par unité de consommation (en euros)











.. tabs::



   




   .. tab:: {"tab_name": "Point"}

      Update a field of a table with the value of "3e décile du revenu déclaré par unité de consommation (en euros)" for a point. The value will be from
      the densest boundary with this measure.

      .. code-block:: postgresql

          WITH data AS (
            SELECT cartodb_id AS id, the_geom FROM <table_to_update>
          ),
          meta AS (
            SELECT OBS_GetMeta(ST_SetSRID(ST_Extent(the_geom), 4326),
            '[{"numer_id": "fr.fr_income.DEC_D312"}]'
            , 1, 1) meta
          FROM data)
          UPDATE <table_to_update> SET <field_to_update> = (data->0->>'value')::Numeric
          FROM OBS_GetData(
                (SELECT ARRAY_AGG((the_geom, id)::geomval) FROM data),
                (SELECT meta FROM meta)) obs_data
          WHERE <table_to_update>.cartodb_id = obs_data.id

      Obtain the value of "3e décile du revenu déclaré par unité de consommation (en euros)" for a point. The value will be from
      the densest boundary with this measure.

      .. code-block:: postgresql

          WITH data AS (
            SELECT 1 AS id,
            ST_SetSRID(ST_Point(2.361, 48.86), 4326) the_geom
          ),
          meta AS (
            SELECT OBS_GetMeta(ST_SetSRID(ST_Extent(the_geom), 4326),
            '[{"numer_id": "fr.fr_income.DEC_D312"}]'
            , 1, 1) meta
          FROM data)
          SELECT id, (data->0->>'value')::Numeric AS dec_d312
          FROM OBS_GetData(
                 (SELECT ARRAY_AGG((the_geom, id)::geomval) FROM data),
                 (SELECT meta FROM meta))
