Rés princ Type maison 1946 à 1990
------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------

Nombre de résidences principales de type maison construites de 1946 à 1990











.. tabs::



   
   .. tab:: {"tab_name": "Per Rés princ type maison"}

      Update a field of a table with an estimate of "Rés princ Type maison 1946 à 1990" per Rés princ type maison for a polygon
      using a combination of weighting by area and Rés princ type maison.

      .. code-block:: postgresql

          WITH data AS (
            SELECT cartodb_id AS id, the_geom FROM <table_to_update>
          ),
          meta AS (
            SELECT OBS_GetMeta(ST_SetSRID(ST_Extent(the_geom), 4326),
            '[{"numer_id": "fr.insee.P12_RPMAISON_ACHT2",
              "denom_id": "fr.insee.P12_RPMAISON",
              "normalization": "denominated"}]'
            , 1, 1) meta
          FROM data)
          UPDATE <table_to_update> SET <field_to_update> = (data->0->>'value')::Numeric
          FROM OBS_GetData(
                (SELECT ARRAY_AGG((the_geom, id)::geomval) FROM data),
                (SELECT meta FROM meta)) obs_data
          WHERE <table_to_update>.cartodb_id = obs_data.id

      Obtain an estimate of "Rés princ Type maison 1946 à 1990" per Rés princ type maison for a polygon
      using a combination of weighting by area and Rés princ type maison.

      .. code-block:: postgresql

          WITH data AS (
            SELECT 1 AS id,
            ST_Buffer(ST_SetSRID(ST_Point(2.361, 48.86), 4326), 0.01) the_geom
          ),
          meta AS (
            SELECT OBS_GetMeta(ST_SetSRID(ST_Extent(the_geom), 4326),
            '[{"numer_id": "fr.insee.P12_RPMAISON_ACHT2",
               "denom_id": "fr.insee.P12_RPMAISON",
               "normalization": "denominated"}]'
            , 1, 1) meta
          FROM data)
          SELECT id, (data->0->>'value')::Numeric AS p12_rpmaison_acht2_rate_2012
          FROM OBS_GetData(
                 (SELECT ARRAY_AGG((the_geom, id)::geomval) FROM data),
                 (SELECT meta FROM meta))
