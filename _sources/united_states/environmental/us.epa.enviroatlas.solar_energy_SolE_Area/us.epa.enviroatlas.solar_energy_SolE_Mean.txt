Annual Average direct normal solar resources kWh/m2/day
------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------













.. tabs::



   
   .. tab:: {"tab_name": "Weighted by Area with solar energy potential "}

      Update a field of a table with an estimate of "Annual Average direct normal solar resources kWh/m2/day" for a polygon
      using a combination of weighting by area and Area with solar energy potential.

      .. code-block:: postgresql

          WITH data AS (
            SELECT cartodb_id AS id, the_geom FROM <table_to_update>
          ),
          meta AS (
            SELECT OBS_GetMeta(ST_SetSRID(ST_Extent(the_geom), 4326),
            '[{"numer_id": "us.epa.enviroatlas.solar_energy_SolE_Mean",
              "denom_id": "us.epa.enviroatlas.solar_energy_SolE_Area"}]'
            , 1, 1) meta
          FROM data)
          UPDATE <table_to_update> SET <field_to_update> = (data->0->>'value')::Numeric
          FROM OBS_GetData(
                (SELECT ARRAY_AGG((the_geom, id)::geomval) FROM data),
                (SELECT meta FROM meta)) obs_data
          WHERE <table_to_update>.cartodb_id = obs_data.id

      Obtain an estimate of "Annual Average direct normal solar resources kWh/m2/day" for a polygon
      using a combination of weighting by area and Area with solar energy potential.

      .. code-block:: postgresql

          WITH data AS (
            SELECT 1 AS id,
            ST_Buffer(ST_SetSRID(ST_Point(-73.9, 40.7), 4326), 0.01) the_geom
          ),
          meta AS (
            SELECT OBS_GetMeta(ST_SetSRID(ST_Extent(the_geom), 4326),
            '[{"numer_id": "us.epa.enviroatlas.solar_energy_SolE_Mean",
               "denom_id": "us.epa.enviroatlas.solar_energy_SolE_Area"}]'
            , 1, 1) meta
          FROM data)
          SELECT id, (data->0->>'value')::Numeric AS sole_mean_estimate_2012
          FROM OBS_GetData(
                 (SELECT ARRAY_AGG((the_geom, id)::geomval) FROM data),
                 (SELECT meta FROM meta))

      

   




   .. tab:: {"tab_name": "Point"}

      Update a field of a table with the value of "Annual Average direct normal solar resources kWh/m2/day" for a point. The value will be from
      the densest boundary with this measure.

      .. code-block:: postgresql

          WITH data AS (
            SELECT cartodb_id AS id, the_geom FROM <table_to_update>
          ),
          meta AS (
            SELECT OBS_GetMeta(ST_SetSRID(ST_Extent(the_geom), 4326),
            '[{"numer_id": "us.epa.enviroatlas.solar_energy_SolE_Mean"}]'
            , 1, 1) meta
          FROM data)
          UPDATE <table_to_update> SET <field_to_update> = (data->0->>'value')::Numeric
          FROM OBS_GetData(
                (SELECT ARRAY_AGG((the_geom, id)::geomval) FROM data),
                (SELECT meta FROM meta)) obs_data
          WHERE <table_to_update>.cartodb_id = obs_data.id

      Obtain the value of "Annual Average direct normal solar resources kWh/m2/day" for a point. The value will be from
      the densest boundary with this measure.

      .. code-block:: postgresql

          WITH data AS (
            SELECT 1 AS id,
            ST_SetSRID(ST_Point(-73.9, 40.7), 4326) the_geom
          ),
          meta AS (
            SELECT OBS_GetMeta(ST_SetSRID(ST_Extent(the_geom), 4326),
            '[{"numer_id": "us.epa.enviroatlas.solar_energy_SolE_Mean"}]'
            , 1, 1) meta
          FROM data)
          SELECT id, (data->0->>'value')::Numeric AS sole_mean
          FROM OBS_GetData(
                 (SELECT ARRAY_AGG((the_geom, id)::geomval) FROM data),
                 (SELECT meta FROM meta))
