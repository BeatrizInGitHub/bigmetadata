Update a field of a table with an estimate of “Three or more room private occupied dwellings” per Occupied private dwellings for a polygon using a combination of weighting by area and Occupied private dwellings.

WITH data AS (
  SELECT cartodb_id AS id, the_geom FROM <table_to_update>
),
meta AS (
  SELECT OBS_GetMeta(ST_SetSRID(ST_Extent(the_geom), 4326),
  '[{"numer_id": "mx.inegi_columns.VIV12",
    "denom_id": "mx.inegi_columns.VIV2",
    "normalization": "denominated"}]'
  , 1, 1) meta
FROM data)
UPDATE <table_to_update> SET <field_to_update> = (data->0->>'value')::Numeric
FROM OBS_GetData(
      (SELECT ARRAY_AGG((the_geom, id)::geomval) FROM data),
      (SELECT meta FROM meta)) obs_data
WHERE <table_to_update>.cartodb_id = obs_data.id

Obtain an estimate of “Three or more room private occupied dwellings” per Occupied private dwellings for a polygon using a combination of weighting by area and Occupied private dwellings.

WITH data AS (
  SELECT 1 AS id,
  ST_Buffer(ST_SetSRID(ST_Point(-99.17, 19.413), 4326), 0.01) the_geom
),
meta AS (
  SELECT OBS_GetMeta(ST_SetSRID(ST_Extent(the_geom), 4326),
  '[{"numer_id": "mx.inegi_columns.VIV12",
     "denom_id": "mx.inegi_columns.VIV2",
     "normalization": "denominated"}]'
  , 1, 1) meta
FROM data)
SELECT id, (data->0->>'value')::Numeric AS three_or_more_room_dwelling_rate_2010
FROM OBS_GetData(
       (SELECT ARRAY_AGG((the_geom, id)::geomval) FROM data),
       (SELECT meta FROM meta))
